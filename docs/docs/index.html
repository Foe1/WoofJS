new Circle({})

Step 3 - Make your circle bigger

setBackdropURL('http://i.imgur.com/q6Bqraw.jpg')

new Circle({radius: 20})

Step 4 - Change your circle's color

setBackdropURL('http://i.imgur.com/q6Bqraw.jpg')

new Circle({radius: 20, color: "yellow"})

Step 5 - Give your circle a name

Step 6 - Make your circle move

setBackdropURL('http://i.imgur.com/q6Bqraw.jpg')

Step 7 - Make your circle move more

setBackdropURL('http://i.imgur.com/q6Bqraw.jpg')
var stevensBall =new Circle({radius: 20, color: "yellow"})
forever(() => {
  stevensBall.move(1)
})

Step 8 - Make your circle follow your mouse

setBackdropURL('http://i.imgur.com/q6Bqraw.jpg')
var stevensBall = new Circle({radius: 20, color: "yellow"})
forever(() => {
  stevensBall.pointTowards(mouseX, mouseY)
  stevensBall.move(1)
})
